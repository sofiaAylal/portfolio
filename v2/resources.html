/**
 * Example function in C++ language
 * Displays the Fibonacci sequence up to n.
 */
#include <iostream>

void fibonacci(int n) {
    int t1 = 0, t2 = 1, nextTerm = 0;
    
    std::cout << "Fibonacci Series: " << t1 << ", " << t2 << ", ";

    for (int i = 3; i <= n; ++i) {
        nextTerm = t1 + t2;
        std::cout << nextTerm << ", ";
        t1 = t2;
        t2 = nextTerm;
    }
}
